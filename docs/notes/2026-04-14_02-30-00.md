Run these as `clusteradmin`. This writes commands to script files and runs them, so you avoid fragile SSH paste behavior.

~~~bash
cat > /tmp/docdown-wheelhouse-setup.sh <<'EOF'
#!/usr/bin/env bash
set -euo pipefail

cat > /tmp/docdown-wheelhouse-user.sh <<'EOS'
#!/usr/bin/env bash
set -euo pipefail

BUILD_ROOT=/home/docdown-runner/wheelhouse-build
SRC_DIR=$BUILD_ROOT/src
WH_DIR=$BUILD_ROOT/wheelhouse
VENV=$BUILD_ROOT/.venv
TEST_VENV=$BUILD_ROOT/test-venv

rm -rf "$BUILD_ROOT"
mkdir -p "$SRC_DIR" "$WH_DIR"

git clone https://github.com/Frank-Yong/DocDown "$SRC_DIR"
cd "$SRC_DIR"

python3 -m venv "$VENV"
"$VENV/bin/pip" install --upgrade pip
"$VENV/bin/pip" download --dest "$WH_DIR" "setuptools>=68.0" wheel
"$VENV/bin/pip" wheel --wheel-dir "$WH_DIR" .

python3 -m venv "$TEST_VENV"
"$TEST_VENV/bin/pip" install --upgrade pip
"$TEST_VENV/bin/pip" install --no-index --find-links "$WH_DIR" "$SRC_DIR"
"$TEST_VENV/bin/docdown" --help >/dev/null
EOS

chmod +x /tmp/docdown-wheelhouse-user.sh
sudo -u docdown-runner -H /bin/bash /tmp/docdown-wheelhouse-user.sh

sudo install -d -o docdown-runner -g docdown-runner /opt/docdown/shared/wheelhouse
sudo rm -rf /opt/docdown/shared/wheelhouse/*
sudo cp -a /home/docdown-runner/wheelhouse-build/wheelhouse/. /opt/docdown/shared/wheelhouse/
sudo chown -R docdown-runner:docdown-runner /opt/docdown/shared/wheelhouse

sudo -u docdown-runner -H /bin/bash -c "ls -1 /opt/docdown/shared/wheelhouse | head -n 20"
sudo -u docdown-runner -H /bin/bash -c "ls -1 /opt/docdown/shared/wheelhouse/docdown-*.whl"
EOF

chmod +x /tmp/docdown-wheelhouse-setup.sh
bash /tmp/docdown-wheelhouse-setup.sh
~~~

Configure passwordless sudo for CD deploy commands:

~~~bash
sudo tee /etc/sudoers.d/docdown-runner >/dev/null <<'EOF'
Defaults:docdown-runner !requiretty
docdown-runner ALL=(root) NOPASSWD: /usr/local/bin/docdown-activate-release *
docdown-runner ALL=(root) NOPASSWD: /usr/bin/systemctl restart docdown-worker
docdown-runner ALL=(root) NOPASSWD: /usr/bin/systemctl restart docdown-api
EOF

sudo chmod 440 /etc/sudoers.d/docdown-runner
sudo visudo -cf /etc/sudoers.d/docdown-runner

sudo -u docdown-runner -H sudo -n /usr/local/bin/docdown-activate-release /opt/docdown/current || true
sudo -u docdown-runner -H sudo -n /usr/bin/systemctl restart docdown-worker
sudo -u docdown-runner -H sudo -n /usr/bin/systemctl restart docdown-api
~~~

Install missing release activation wrapper (fixes `/usr/local/bin/docdown-activate-release: command not found`):

~~~bash
sudo tee /usr/local/bin/docdown-activate-release >/dev/null <<'EOF'
#!/usr/bin/env bash
set -euo pipefail

release_dir="${1:-}"
if [ -z "${release_dir}" ]; then
	echo "Usage: docdown-activate-release <release-path>" >&2
	exit 2
fi

if [ ! -d "${release_dir}" ]; then
	echo "Release path does not exist or is not a directory: ${release_dir}" >&2
	exit 1
fi

ln -sfn "${release_dir}" /opt/docdown/current
EOF

sudo chmod 755 /usr/local/bin/docdown-activate-release
sudo chown root:root /usr/local/bin/docdown-activate-release
sudo /usr/local/bin/docdown-activate-release "$(ls -1dt /opt/docdown/releases/* | head -n 1)"
~~~
